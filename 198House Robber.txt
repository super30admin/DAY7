198. House Robber

class Solution {
    //TC=O(n) SC=O(n)
    public int rob(int[] nums) {
    int n= nums.length;
    if(n<2){
        return nums[0];
    }
    int [] dp=new int[nums.length];
    dp[0]=nums[0];
    dp[1]=Math.max(nums[0],nums[1]);

    for(int i=2;i<nums.length;i++){
        dp[i]=Math.max(dp[i-1],nums[i]+dp[i-2]);
    }
        return dp[n-1];
    }

}
